## Kotlin contract (already shipped by Plans 15.1-01 through 15.1-03)

The shared Kotlin side exposes:

```kotlin
// shared/src/commonMain/kotlin/com/pumpernickel/presentation/gamification/RanksAndAchievementsViewModel.kt
class RanksAndAchievementsViewModel(gamificationRepository: GamificationRepository) : ViewModel() {
    @NativeCoroutinesState
    val rankLadderState: StateFlow<RankLadderUiState>
}

data class RankLadderUiState(
    val rows: List<RankRow> = emptyList(),
    val totalXp: Long = 0L,
    val currentRank: Rank? = null,        // null when Unranked
    val isUnranked: Boolean = true,
    val isLoading: Boolean = true
)

data class RankRow(
    val rank: Rank,
    val threshold: Long,
    val status: RankRowStatus,
    val xpToReach: Long?
)

enum class RankRowStatus { PASSED, CURRENT, LOCKED }
```

Swift sees this as:

- `Shared.RanksAndAchievementsViewModel` with generated property `rankLadderStateFlow` (observable via `asyncSequence(for:)`).
- `Shared.RankLadderUiState` — data class, field access via `state.rows`, `state.totalXp`, `state.currentRank`, `state.isUnranked`, `state.isLoading`.
- `Shared.RankRow` — `row.rank`, `row.threshold`, `row.status`, `row.xpToReach` (the last bridges as `KotlinLong?`).
- `Shared.RankRowStatus` enum cases as `.passed`, `.current`, `.locked` (Kotlin-Native objc convention lowercases + camelCases the case name).

### VM retrieval (Swift)

```swift
private let viewModel = RanksAndAchievementsKoinHelper().getRanksAndAchievementsViewModel()
```

Note: `RanksAndAchievementsKoinHelper` is a **class**, not an object. Swift call signature is `ClassName().method()` — an empty-parens constructor then the method. This differs from the root `KoinHelper` which is an `object` and is called as `KoinHelper.shared.getFooViewModel()`. Mirrors `GamificationUiKoinHelper()` and `AchievementGalleryKoinHelper()` (both class-style helpers from Phase 15-08/09).

## File 1 — `iosApp/iosApp/Views/Overview/RankLadderView.swift` (NEW)

Observes `rankLadderStateFlow`, renders a `List` or `ScrollView + LazyVStack` of 10 rank rows per `RankLadderUiState.rows`, with PASSED / CURRENT / LOCKED presentation. Use `NumberFormatter.localizedString(from: number, number: .decimal)` for the XP threshold (D-151-18).

### Template (copy + polish to taste)

```swift
import SwiftUI
import Shared
import KMPNativeCoroutinesAsync

struct RankLadderView: View {
    private let viewModel = RanksAndAchievementsKoinHelper().getRanksAndAchievementsViewModel()

    @State private var uiState: SharedRankLadderUiState?

    var body: some View {
        Group {
            if let state = uiState, !state.isLoading {
                List {
                    if state.isUnranked {
                        unrankedHeader
                            .listRowSeparator(.hidden)
                            .listRowBackground(Color.clear)
                    }
                    ForEach(rows(from: state), id: \.rank) { row in
                        RankLadderRow(row: row, isUnranked: state.isUnranked)
                            .listRowSeparator(.hidden)
                    }
                }
                .listStyle(.plain)
            } else {
                ProgressView()
                    .frame(maxWidth: .infinity, maxHeight: .infinity)
            }
        }
        .navigationTitle("Ranks")
        .navigationBarTitleDisplayMode(.inline)
        .task { await observe() }
    }

    private var unrankedHeader: some View {
        HStack(spacing: 12) {
            Image(systemName: "lock.fill")
                .font(.title3)
                .foregroundColor(.secondary)
            // D-151-23 / D-11: load-bearing literal. Same em-dash (U+2014) as
            // OverviewRankStrip.swift line ~45. DO NOT localise or rephrase.
            Text("Unranked — complete a workout to unlock Silver")
                .font(.subheadline)
                .foregroundColor(.secondary)
            Spacer()
        }
        .padding(.vertical, 8)
    }

    private func rows(from state: SharedRankLadderUiState) -> [SharedRankRow] {
        // Kotlin List<RankRow> imports as `NSArray` — bridge to Swift array.
        guard let arr = state.rows as? [SharedRankRow] else { return [] }
        return arr
    }

    private func observe() async {
        do {
            for try await state in asyncSequence(for: viewModel.rankLadderStateFlow) {
                self.uiState = state
            }
        } catch {
            print("RankLadder observation error: \(error)")
        }
    }
}

// MARK: - Row

private struct RankLadderRow: View {
    let row: SharedRankRow
    let isUnranked: Bool

    var body: some View {
        HStack(spacing: 12) {
            Image(systemName: "rosette")
                .font(.system(size: 28, weight: .bold))
                .foregroundColor(tint(row.rank))
                .frame(width: 40)

            VStack(alignment: .leading, spacing: 2) {
                Text("#\(row.rank.tier)  \(row.rank.displayName)")
                    .font(.headline)
                Text("\(formattedThreshold) XP")
                    .font(.caption)
                    .foregroundColor(.secondary)
                HStack(spacing: 6) {
                    Image(systemName: statusIcon).font(.caption2)
                    Text(statusText).font(.caption2)
                }
                .foregroundColor(statusTint)
            }

            Spacer()
        }
        .padding(.vertical, 6)
        .opacity(row.status == SharedRankRowStatus.locked ? 0.45 : 1.0)
    }

    private var formattedThreshold: String {
        // D-151-18: locale-aware thousands separator.
        NumberFormatter.localizedString(
            from: NSNumber(value: row.threshold),
            number: .decimal
        )
    }

    private var statusIcon: String {
        switch row.status {
        case .passed:  return "checkmark.circle.fill"
        case .current: return "star.fill"
        case .locked:  return "lock.fill"
        default:       return "lock.fill"
        }
    }

    private var statusText: String {
        switch row.status {
        case .passed:  return "Passed"
        case .current: return "Current rank"
        case .locked:
            let xpToReach = row.xpToReach?.int64Value ?? 0
            if isUnranked && row.rank == SharedRank.silver && xpToReach == 0 {
                // D-151-07: SILVER when unranked → first-workout hint.
                return "First workout unlocks"
            }
            let formatted = NumberFormatter.localizedString(
                from: NSNumber(value: xpToReach),
                number: .decimal
            )
            return "\(formatted) XP to unlock"
        default: return ""
        }
    }

    private var statusTint: Color {
        switch row.status {
        case .passed, .current: return .accentColor
        default: return .secondary
        }
    }

    /// Tier-banded tint matching OverviewRankStrip.swift `tint(for:)` and
    /// UnlockModalView.swift `rankTint` — keep them in sync.
    private func tint(_ rank: SharedRank) -> Color {
        switch rank.tier {
        case 1, 2:    return Color(red: 0.75, green: 0.75, blue: 0.78)
        case 3, 4, 5: return Color(red: 1.00, green: 0.84, blue: 0.20)
        case 6, 7:    return Color(red: 0.60, green: 0.80, blue: 1.00)
        case 8, 9:    return Color(red: 0.73, green: 0.41, blue: 0.78)
        default:      return Color(red: 1.00, green: 0.25, blue: 0.25)
        }
    }
}

// MARK: - Shared type aliases
// Note: KMP-Native exports sealed subclasses as NESTED Swift types
// (Shared.RankState.Unranked), not flat (Shared.RankStateUnranked). This
// was discovered during Phase 15 and is documented at the top of
// OverviewRankStrip.swift:118-125.

private typealias SharedRankLadderUiState = Shared.RankLadderUiState
private typealias SharedRankRow = Shared.RankRow
private typealias SharedRankRowStatus = Shared.RankRowStatus
private typealias SharedRank = Shared.Rank
```

### Rules / gotchas

1. `RankRowStatus` enum cases in Swift are lowercased — `.passed`, `.current`, `.locked`. If Xcode autocomplete shows `.PASSED`, adjust. Run one build to confirm.
2. `row.xpToReach` is `KotlinLong?` — bridge via `.int64Value` when non-nil. See template.
3. The `NumberFormatter.localizedString(from:number:)` approach is preferred over string interpolation so users get their locale's thousands separator for big numbers.
4. If the Swift compiler rejects `state.rows as? [SharedRankRow]`, try `state.rows as NSArray as? [SharedRankRow]` or iterate `for case let r as SharedRankRow in state.rows` — the Kotlin List → Swift interop shape varies slightly across Kotlin versions.
5. The D-11 literal `"Unranked — complete a workout to unlock Silver"` MUST match `OverviewRankStrip.swift`'s version character-for-character (em dash U+2014, single spaces each side).

/// D-151-12 container — a single scrollable composition with the rank ladder
/// on top and a NavigationLink to the existing achievements gallery at the

struct RanksAndAchievementsView: View {

            VStack(spacing: 16) {

                // a helper and call it from both places. For first pass, a

                    .cornerRadius(12)

                .padding(.horizontal)
            }
            .padding(.vertical)
        }
        .navigationTitle("Ranks & Achievements")
        .navigationBarTitleDisplayMode(.inline)

}

/// Inline variant of the ladder for use inside the container above — same
/// logic, without the Scaffold-style nav wrapper so it can embed cleanly.
private struct RankLadderInline: View {
    // Copy RankLadderView's VM retrieval + .task + body logic here, OR
    // factor both into a shared helper. Reasonable first pass: just call
    // RankLadderView() — SwiftUI will present its .navigationTitle on the
    // parent's nav bar (harmless), and the List inside renders inline.

        RankLadderView()
            .frame(minHeight: 700)   // reserve height; a List inside a ScrollView needs an explicit min
    }

### Rules / gotchas

1. If `List inside ScrollView` causes layout issues, replace `List` in `RankLadderView` with `LazyVStack` and drop `listRowSeparator` calls.
2. The `AchievementGalleryView()` NavigationLink reaches the existing Phase 15-10 view — no import needed beyond `Shared`.
3. `Color(.secondarySystemGroupedBackground)` is the consistent card background across the iOS app (OverviewView.swift uses it; AchievementGalleryView.swift uses it).
4. If the nested `.navigationTitle` causes double titles, delete the inner `.navigationTitle("Ranks")` from `RankLadderView` when it's embedded in this container (simplest: add a `showsTitle: Bool = true` parameter to `RankLadderView` and gate the modifier on it).
