**CRITICAL: How to extract paths from Svelte files:**
1. Open AnatomyFront.svelte. Find `<path>` elements inside `<g>` elements.
2. Each interactive muscle region path has a CSS class matching a key in muscleRegionMap (e.g., `class="upper-pectoralis"`).
3. Some regions have multiple `<path>` elements (e.g., left and right pectoralis are separate paths but both have class "upper-pectoralis").
4. Copy the FULL `d=""` value for each path. These are long strings (hundreds of characters each).
5. Non-interactive elements (body outline, hands, feet) have different classes -- extract these separately for the outline layer.
6. Repeat for AnatomyBack.svelte.

**IMPORTANT:** The SVG paths MUST be extracted verbatim from the Svelte files. Do NOT attempt to approximate or simplify them. Each path's `d` attribute is the exact shape data.

**Step 2: AnatomyFrontShape.swift** -- SwiftUI rendering of front body:

Use nicklockwood/SVGPath library (added via SPM) to parse path strings into SwiftUI Path objects:
```swift
import SwiftUI
import SVGPath  // nicklockwood/SVGPath SPM package

struct AnatomyFrontView: View {
    let selectedGroup: String?  // MuscleGroup.dbName or nil
    let onRegionTapped: (String) -> Void  // Returns group name

    var body: some View {
        GeometryReader { geo in
            let scale = geo.size.width / MuscleRegionPaths.viewBoxWidth
            let scaledHeight = MuscleRegionPaths.viewBoxHeight * scale

            ZStack {
                // Body outline (non-interactive)
                ForEach(MuscleRegionPaths.frontOutline.indices, id: \.self) { i in
                    SVGPathShape(pathData: MuscleRegionPaths.frontOutline[i])
                        .stroke(Color(white: 0.165), lineWidth: 1)  // #2A2A2A dark mode stroke
                        .scaleEffect(x: scale, y: scale, anchor: .topLeading)
                }

                // Interactive muscle regions
                ForEach(MuscleRegionPaths.frontRegions, id: \.regionId) { region in
                    let isSelected = region.groupName == selectedGroup
                    SVGPathShape(pathData: region.pathData)
                        .fill(isSelected
                            ? Color(red: 0.4, green: 0.733, blue: 0.416).opacity(0.8)  // #66BB6A at 80%
                            : Color(white: 0.227))  // #3A3A3A idle
                        .scaleEffect(x: scale, y: scale, anchor: .topLeading)
                        .onTapGesture {
                            onRegionTapped(region.groupName)
                        }
                }
            }
            .frame(height: scaledHeight)
        }
        .aspectRatio(MuscleRegionPaths.viewBoxWidth / MuscleRegionPaths.viewBoxHeight, contentMode: .fit)
    }
}

// Helper Shape wrapper for SVG path data
struct SVGPathShape: Shape {
    let pathData: String

    func path(in rect: CGRect) -> Path {
        // Use SVGPath library to parse
        return Path(svgPath: pathData)
    }
}
```

struct AnatomyPickerView: View {
    @Binding var selectedGroup: String?  // nil = no selection
    let onConfirm: (String) -> Void
    @Environment(\.dismiss) private var dismiss

    @State private var localSelection: String? = nil

    var body: some View {
        NavigationStack {
            VStack(spacing: 16) {
                // Front and back silhouettes side by side
                HStack(spacing: 16) {
                    AnatomyFrontView(
                        selectedGroup: localSelection,
                        onRegionTapped: { group in localSelection = group }
                    )
                    AnatomyBackView(
                        selectedGroup: localSelection,
                        onRegionTapped: { group in localSelection = group }
                    )
                }
                .padding(.horizontal, 16)

                // Selected muscle group label
                if let group = localSelection {
                    Text(displayName(for: group))
                        .font(.title3.weight(.semibold))  // Heading style
                } else {
                    Text("Tap a muscle group")
                        .font(.title3)
                        .foregroundColor(.secondary)
                }

                Spacer()

                // Confirm button
                Button(action: {
                    if let group = localSelection {
                        onConfirm(group)
                        dismiss()
                    }
                }) {
                    Text("Select")
                        .font(.body.weight(.semibold))
                        .foregroundColor(.white)
                        .frame(maxWidth: .infinity)
                        .frame(height: 48)
                        .background(localSelection != nil
                            ? Color(red: 0.4, green: 0.733, blue: 0.416)
                            : Color.gray)
                        .cornerRadius(12)
                }
                .disabled(localSelection == nil)
                .padding(.horizontal, 32)
                .padding(.bottom, 16)
            }
            .navigationTitle("Select Muscle Group")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("", systemImage: "xmark") { dismiss() }
                }
            }
        }
        .onAppear { localSelection = selectedGroup }
    }

    private func displayName(for groupDbName: String) -> String {
        // Map dbName back to display name
        // Use the MuscleGroup enum from the Shared framework
        // e.g., MuscleGroup.companion.fromDbName(name: groupDbName)?.displayName ?? groupDbName.capitalized
        groupDbName.capitalized  // Fallback if Kotlin enum not easily accessible
    }
}

struct CreateExerciseView: View {
    @Environment(\.dismiss) private var dismiss
    // Get CreateExerciseViewModel from Koin
    @State private var name = ""
    @State private var selectedMuscleGroup: String? = nil  // dbName
    @State private var selectedEquipment: String? = nil
    @State private var selectedCategory: String? = nil
    @State private var equipmentOptions: [String] = []
    @State private var categoryOptions: [String] = []
    @State private var isFormValid = false
    @State private var showAnatomyPicker = false
    @State private var showSuccessToast = false

    var body: some View {
        VStack(spacing: 24) {  // lg spacing between fields
            // Name field
            TextField("Exercise name", text: $name)
                .font(.body)
                .padding(16)  // md padding
                .background(Color(white: 0.12))  // #1E1E1E secondary
                .cornerRadius(8)
                .onChange(of: name) { _, newValue in
                    viewModel.onNameChanged(name: newValue)
                }

            // Muscle group field (tap opens anatomy picker)
            Button(action: { showAnatomyPicker = true }) {
                HStack {
                    Text(selectedMuscleGroup != nil
                        ? displayName(for: selectedMuscleGroup!)
                        : "Select muscle group")
                        .foregroundColor(selectedMuscleGroup != nil ? .primary : .secondary)
                    Spacer()
                    Image(systemName: "chevron.right")
                        .foregroundColor(.secondary)
                }
                .font(.body)
                .padding(16)
                .background(Color(white: 0.12))
                .cornerRadius(8)
            }

            // Equipment picker
            Picker("Equipment", selection: $selectedEquipment) {
                Text("Select equipment type").tag(nil as String?)
                ForEach(equipmentOptions, id: \.self) { option in
                    Text(option).tag(option as String?)
                }
            }
            .pickerStyle(.menu)  // iOS native dropdown

            // Category picker
            Picker("Category", selection: $selectedCategory) {
                Text("Select a category").tag(nil as String?)
                ForEach(categoryOptions, id: \.self) { option in
                    Text(option).tag(option as String?)
                }
            }
            .pickerStyle(.menu)

            Spacer()

            // Create button
            Button(action: {
                viewModel.createExercise()
            }) {
                Text("Create Exercise")
                    .font(.body.weight(.semibold))
                    .foregroundColor(.white)
                    .frame(maxWidth: .infinity)
                    .frame(height: 48)
                    .background(isFormValid
                        ? Color(red: 0.4, green: 0.733, blue: 0.416)
                        : Color(red: 0.4, green: 0.733, blue: 0.416).opacity(0.5))
                    .cornerRadius(12)
            }
            .disabled(!isFormValid)
        }
        .padding(.horizontal, 16)
        .padding(.top, 16)
        .navigationTitle("New Exercise")
        .sheet(isPresented: $showAnatomyPicker) {
            AnatomyPickerView(
                selectedGroup: $selectedMuscleGroup,
                onConfirm: { group in
                    selectedMuscleGroup = group
                    // Convert group dbName to MuscleGroup enum and pass to VM
                    viewModel.onMuscleGroupSelected(group: MuscleGroup.companion.fromDbName(name: group))
                }
            )
            .presentationDetents([.large])
        }
        .overlay(alignment: .top) {
            if showSuccessToast {
                Text("Exercise created")
                    .font(.subheadline)
                    .padding(.horizontal, 16)
                    .padding(.vertical, 10)
                    .background(.ultraThinMaterial)
                    .cornerRadius(8)
                    .transition(.move(edge: .top).combined(with: .opacity))
                    .padding(.top, 8)
            }
        }
        .task {
            // Observe equipment/category options and saveResult from ViewModel
            // On SaveResult.Success: show toast for 2s, then dismiss
        }
    }
}
