**2. TemplateEditorView.swift** (per D-05, D-06, D-07, D-08, D-10, D-12):

The template create/edit screen with inline target configuration and drag-and-drop reordering.

IMPORTANT: Do NOT define a local `isFormValid` computed property that duplicates the ViewModel logic. Instead, observe the ViewModel's `isFormValid` StateFlow via `asyncSequence` and use the `@State private var isFormValid` variable to drive the Save button's disabled state. This keeps validation logic in a single source of truth (the ViewModel) and ensures the SwiftUI view matches the KMP validation exactly.

```swift
import SwiftUI
import Shared
import KMPNativeCoroutinesAsync

struct TemplateEditorView: View {
    let templateId: Int64?

    private let viewModel = KoinHelper.shared.getTemplateEditorViewModel()

    @Environment(\.dismiss) private var dismiss

    @State private var name: String = ""
    @State private var exercises: [TemplateExercise] = []
    @State private var isSaving: Bool = false
    @State private var isFormValid: Bool = false  // Observed from ViewModel, NOT computed locally
    @State private var saveResult: TemplateEditorViewModel.SaveResult? = nil
    @State private var showExercisePicker = false
    @State private var showErrorAlert = false
    @State private var errorMessage = ""

    init(templateId: Int64? = nil) {
        self.templateId = templateId
    }

    var body: some View {
        VStack(spacing: 0) {
            Form {
                // Template name section
                Section("Template Name") {
                    TextField("e.g., Push Day", text: $name)
                        .onChange(of: name) { _, newValue in
                            viewModel.onNameChanged(name: newValue)
                        }
                }

                // Exercises section with inline targets
                Section {
                    if exercises.isEmpty {
                        Text("No exercises added yet")
                            .foregroundColor(.secondary)
                            .font(.subheadline)
                    } else {
                        ForEach(Array(exercises.enumerated()), id: \.element.id) { index, exercise in
                            exerciseRow(exercise: exercise)
                        }
                        .onMove { source, destination in
                            if let from = source.first {
                                viewModel.moveExercise(from: Int32(from), to: Int32(destination))
                            }
                        }
                        .onDelete { indices in
                            if let index = indices.first, index < exercises.count {
                                viewModel.removeExercise(templateExerciseId: exercises[index].id)
                            }
                        }
                    }
                } header: {
                    HStack {
                        Text("Exercises")
                        Spacer()
                        Button("Add Exercise") {
                            showExercisePicker = true
                        }
                        .font(.subheadline)
                    }
                }
            }
            .environment(\.editMode, .constant(.active)) // Always show drag handles (D-12)
        }
        .navigationTitle(templateId != nil ? "Edit Template" : "New Template")
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .navigationBarTrailing) {
                Button("Save") {
                    viewModel.save()
                }
                .disabled(!isFormValid || isSaving)
            }
        }
        .sheet(isPresented: $showExercisePicker) {
            ExercisePickerView { exerciseId, exerciseName, primaryMuscles in
                viewModel.addExercise(
                    exerciseId: exerciseId,
                    exerciseName: exerciseName,
                    primaryMuscles: primaryMuscles
                )
            }
        }
        .alert("Error", isPresented: $showErrorAlert) {
            Button("OK", role: .cancel) {}
        } message: {
            Text(errorMessage)
        }
        .task {
            if let id = templateId {
                viewModel.loadTemplate(id: id)
            }
            await observeState()
        }
    }

    // MARK: - Exercise Row with Inline Targets (D-05)
    @ViewBuilder
    private func exerciseRow(exercise: TemplateExercise) -> some View {
        VStack(alignment: .leading, spacing: 8) {
            // Exercise name and primary muscle
            HStack {
                Text(exercise.exerciseName)
                    .font(.body.weight(.semibold))
                Spacer()
            }

            if let firstMuscle = exercise.primaryMuscles.first {
                Text(firstMuscle.displayName)
                    .font(.caption)
                    .foregroundColor(.secondary)
            }

            // Inline target configuration (D-05)
            HStack(spacing: 12) {
                targetField(
                    label: "Sets",
                    value: Int(exercise.targetSets),
                    onChange: { newSets in
                        viewModel.updateExerciseTargets(
                            id: exercise.id,
                            sets: Int32(newSets),
                            reps: exercise.targetReps,
                            weightKgX10: exercise.targetWeightKgX10,
                            restSec: exercise.restPeriodSec
                        )
                    }
                )

                targetField(
                    label: "Reps",
                    value: Int(exercise.targetReps),
                    onChange: { newReps in
                        viewModel.updateExerciseTargets(
                            id: exercise.id,
                            sets: exercise.targetSets,
                            reps: Int32(newReps),
                            weightKgX10: exercise.targetWeightKgX10,
                            restSec: exercise.restPeriodSec
                        )
                    }
                )

                weightField(exercise: exercise)

                restField(exercise: exercise)
            }
        }
        .padding(.vertical, 4)
    }

    // MARK: - Target Input Fields
    private func targetField(label: String, value: Int, onChange: @escaping (Int) -> Void) -> some View {
        VStack(spacing: 2) {
            Text(label)
                .font(.caption2)
                .foregroundColor(.secondary)
            TextField("", value: .constant(value), format: .number)
                .keyboardType(.numberPad)
                .multilineTextAlignment(.center)
                .frame(width: 50)
                .padding(.vertical, 6)
                .background(Color(white: 0.12))
                .cornerRadius(8)
                .onChange(of: value) { _, newValue in
                    onChange(newValue)
                }
        }
    }

    // Weight field: display as kg with one decimal (D-06)
    private func weightField(exercise: TemplateExercise) -> some View {
        VStack(spacing: 2) {
            Text("kg")
                .font(.caption2)
                .foregroundColor(.secondary)
            let displayValue = formatWeightDisplay(kgX10: Int(exercise.targetWeightKgX10))
            TextField("0", text: .constant(displayValue))
                .keyboardType(.decimalPad)
                .multilineTextAlignment(.center)
                .frame(width: 60)
                .padding(.vertical, 6)
                .background(Color(white: 0.12))
                .cornerRadius(8)
        }
    }

    // Rest field: display as seconds (D-07)
    private func restField(exercise: TemplateExercise) -> some View {
        VStack(spacing: 2) {
            Text("Rest")
                .font(.caption2)
                .foregroundColor(.secondary)
            TextField("", value: .constant(Int(exercise.restPeriodSec)), format: .number)
                .keyboardType(.numberPad)
                .multilineTextAlignment(.center)
                .frame(width: 50)
                .padding(.vertical, 6)
                .background(Color(white: 0.12))
                .cornerRadius(8)
        }
    }

    // MARK: - Helpers
    private func formatWeightDisplay(kgX10: Int) -> String {
        let whole = kgX10 / 10
        let decimal = kgX10 % 10
        if decimal == 0 {
            return "\(whole)"
        } else {
            return "\(whole).\(decimal)"
        }
    }

    // MARK: - Flow Observation
    private func observeState() async {
        await withTaskGroup(of: Void.self) { group in
            group.addTask { await observeName() }
            group.addTask { await observeExercises() }
            group.addTask { await observeSaving() }
            group.addTask { await observeSaveResult() }
            group.addTask { await observeIsFormValid() }
        }
    }

    private func observeName() async {
        do {
            for try await value in asyncSequence(for: viewModel.name) {
                if self.name != value { self.name = value }
            }
        } catch { print("Name observation error: \(error)") }
    }

    private func observeExercises() async {
        do {
            for try await value in asyncSequence(for: viewModel.exercises) {
                self.exercises = value
            }
        } catch { print("Exercises observation error: \(error)") }
    }

    private func observeSaving() async {
        do {
            for try await value in asyncSequence(for: viewModel.isSaving) {
                self.isSaving = value as! Bool
            }
        } catch { print("Saving observation error: \(error)") }
    }

    private func observeSaveResult() async {
        do {
            for try await value in asyncSequence(for: viewModel.saveResult) {
                if let result = value {
                    switch result {
                    case let success as TemplateEditorViewModel.SaveResultSuccess:
                        dismiss()
                    case let error as TemplateEditorViewModel.SaveResultError:
                        errorMessage = error.message
                        showErrorAlert = true
                    default:
                        break
                    }
                    viewModel.clearSaveResult()
                }
            }
        } catch { print("SaveResult observation error: \(error)") }
    }

    // Observe ViewModel.isFormValid instead of computing locally.
    // This keeps validation in a single source of truth (the ViewModel).
    private func observeIsFormValid() async {
        do {
            for try await value in asyncSequence(for: viewModel.isFormValid) {
                self.isFormValid = value as! Bool
            }
        } catch {
            // Fallback: if Boolean StateFlow bridging fails (known Phase 1 issue),
            // compute locally as a safety net.
            print("isFormValid observation error: \(error)")
        }
    }
}
```

CRITICAL IMPLEMENTATION NOTES for the executor:

1. **Kotlin sealed class in Swift**: `TemplateEditorViewModel.SaveResult` is a Kotlin sealed class. In Swift via KMP, subclasses are accessed as `TemplateEditorViewModel.SaveResultSuccess` and `TemplateEditorViewModel.SaveResultError` (flattened naming). Verify the exact names by checking the generated Shared framework headers. If the naming convention differs, adapt the `switch` accordingly.

2. **Boolean StateFlow observation issue**: Phase 1 discovered that `@NativeCoroutinesState` on `StateFlow<Boolean>` may not bridge correctly via `asyncSequence`. The `observeIsFormValid()` method has a catch block that prints a fallback message. If Boolean observation fails at runtime, the executor should fall back to a local computed property as a safety net. But try the ViewModel observation first -- it is the correct architecture.

3. **Target field editing**: The `.constant()` bindings shown above are placeholders for the pattern. The actual implementation needs mutable @State-backed bindings or a custom approach. A practical approach: use `@State` local copies initialized from the exercise values, with `.onSubmit` or `.onChange` calling the ViewModel. Alternatively, wrap each exercise row in a sub-view with its own @State for the text fields. The executor should implement working two-way text field binding for sets, reps, weight, and rest that calls `viewModel.updateExerciseTargets()` when the user finishes editing.

4. **Weight input (D-06)**: User enters a decimal like "72.5". Parse it to kgX10 (725) before calling updateExerciseTargets. Use the `parseWeightKgX10` function from the Kotlin domain model, or reimplement in Swift: `Int(doubleValue * 10)`.

5. **.onMove + .onDelete together**: Both modifiers work on the same ForEach. With `editMode = .constant(.active)`, drag handles show. The `.onDelete` provides swipe-to-delete within the form.

6. **ExercisePickerView sheet**: Presented as a `.sheet()`. When the user selects an exercise, the sheet dismisses and calls the onSelect closure with exerciseId, exerciseName, and primaryMuscles.
